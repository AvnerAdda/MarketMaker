$ mkdir ~/MarketMaker (creer un  dossier MarkerMaker)
$ cd MarketMaker (rentre dans MarketMaker)
$ git init (initialisation d'un depot)
$ touch READ (creer un fichier READ)
$ git add READ (ajoute READ au depot)
$ git commit -m 'premier commit' (commiter un fichier avec le message 'premier comit')
$ git remote add origin https://github.com/AvnerAdda/MarketMaker.git (ajoute au projet MarketMaker)
$ git push origin master (ajouter)
$ git pull (ajoute a une nouvll branche)
$ cd .. (retourner au fichier precedent)
$ ls (affiche les fichier en local)
$ rm -rf .git (s'enlever de master)
$ git clone git://git.kernel.org/pub/scm/git/git.git (mettre a jour github)
$ git config (mes config)
$ git config --list (toute les config)
$ git push --force --set-upstream origin master (verification)
